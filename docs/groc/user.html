User API

var restify = require('restify'), mongoPersister = require('../mongodb-persister/mongoPersister');

GET /user/:userId

  • PARAM app : application name
  • URI userId : user id
exports.retrieve = function (req, res, next) { mongoPersister.retrieveUser( req.params.app, req.params.userId, function (err, user, msg) { if(err) next(err); if(user){ res.send(user); }else{ res.send(msg); } }); };

POST /user/register

  • PARAM app : application name
  • PARAM userId : user id
  • PARAM deviceType : web / android / iphone
  • PARAM deviceId : device id (unique key) (optional)
  • PARAM notiId : notification token key (optional)
  • PARAM datas : additional data (JSON Object) (optional)
exports.register = function (req, res, next) { console.log(req.params); var _param = req.params; if (_param.app === undefined) { return next(new restify.InvalidArgumentError('App must be supplied')) } if (_param.userId === undefined) { return next(new restify.InvalidArgumentError('User Id must be supplied')) } if(_param.deviceType === undefined) { return next(new restify.InvalidArgumentError('Device Type must be supplied')) } mongoPersister.registerUser( _param.app, _param.userId, _param.deviceType, _param.deviceId, _param.notiId, _param.datas, function (err, msg) { if(err) next(err); res.send(msg); }); };

POST /user/login

  • PARAM app : application name
  • PARAM userId : user id
exports.login = function (req, res, next) { var _param = req.params; if (_param.app === undefined) { return next(new restify.InvalidArgumentError('App must be supplied')) } if (_param.userId === undefined) { return next(new restify.InvalidArgumentError('User Id must be supplied')) } mongoPersister.createUserSessionId(_param.app, _param.userId, function (err, sessionId) { if(err) next(err); res.send({'sessionId': sessionId}); }); };

POST /user/logout

  • PARAM app : application name
  • PARAM userId : user id
exports.logout = function (req, res, next) { var _param = req.params; if (_param.app === undefined) { return next(new restify.InvalidArgumentError('App must be supplied')) } if (_param.userId === undefined) { return next(new restify.InvalidArgumentError('User Id must be supplied')) } mongoPersister.removeUserSessionId(_param.app, _param.userId, function (err) { if(err) next(err); res.send({message:'success'}); }); };
